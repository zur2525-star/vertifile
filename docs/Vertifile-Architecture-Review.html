1. Client-Side Security — Zero-Knowledge Encryption

Documents uploaded to Vertifile are encrypted entirely in the user's browser before any data is transmitted to the server. The server receives only an encrypted blob and never has access to the plaintext content or the decryption key. All cryptographic operations use the Web Crypto API (SubtleCrypto) with no external libraries.

1.1 AES-256-GCM Key Generation

File: public/js/crypto.js — Lines 96-109
async function generateKey() {
  return await subtle.generateKey(
    { name: 'AES-GCM', length: 256 },
    true,              // extractable — needed for URL fragment export
    ['encrypt', 'decrypt']
  );
}

1.2 SHA-256 Hash of Original Content

Before encryption, a SHA-256 hash of the original plaintext is computed. This hash is sent to the server for integrity binding and is later used to verify successful decryption.

File: public/js/crypto.js — Lines 183-196
async function hashContent(data) {
  var hashBuffer = await subtle.digest('SHA-256', data);
  var hashArray = Array.from(new Uint8Array(hashBuffer));
  return hashArray.map(function (b) {
    return b.toString(16).padStart(2, '0');
  }).join('');
}

1.3 File Encryption Flow

The encryptFile() function orchestrates the full client-side encryption: hash the original, generate a key, encrypt with a random 12-byte IV, and export the key as base64url.

File: public/js/crypto.js — Lines 267-283
async function encryptFile(file) {
  var plaintext = await file.arrayBuffer();
  var hash = await hashContent(plaintext);         // SHA-256 of ORIGINAL
  var key = await generateKey();                    // AES-256-GCM key
  var result = await encrypt(key, plaintext);       // Encrypt with random 12-byte IV
  var keyBase64url = await exportKey(key);
  return {
    encryptedBlob: new Blob([result.ciphertext],
      { type: 'application/octet-stream' }),
    hash: hash,                                     // Sent to server
    iv: arrayBufferToBase64url(result.iv),           // Sent to server
    keyBase64url: keyBase64url                       // NEVER sent to server
  };
}
Security Property: The keyBase64url field is returned to the calling code but is NEVER included in the FormData sent to the server. It is only embedded in the URL fragment on the client side.

1.5 HMAC Signing (Server-Side, Layer 2)

File: services/pvf-generator.js — Lines 43-54
function hashBytes(buffer) {
  return crypto.createHash('sha256').update(buffer).digest('hex');
}

function signHash(hash) {
  return crypto.createHmac('sha256', HMAC_SECRET).update(hash).digest('hex');
}

function verifySignature(hash, signature) {
  const expected = signHash(hash);
  return crypto.timingSafeEqual(Buffer.from(expected), Buffer.from(signature));
}

1.6 Ed25519 Dual Signature (Server-Side, Layer 3)

File: services/signing.js — Lines 104-138
async function signEd25519(payload) {
  const primary = await keyManager.getActivePrimary();
  if (!primary) return null;
  const buf = Buffer.isBuffer(payload)
    ? payload : Buffer.from(String(payload), 'utf8');
  const sig = crypto.sign(null, buf, primary.privateKey);
  return { signature: sig.toString('base64url'), keyId: primary.keyId };
}

async function verifyEd25519(payload, signatureBase64url, keyId) {
  if (typeof signatureBase64url !== 'string' || !signatureBase64url) return false;
  if (typeof keyId !== 'string' || !keyId) return false;
  const publicKey = await keyManager.getPublicKeyById(keyId);
  if (!publicKey) return false;
  const buf = Buffer.isBuffer(payload)
    ? payload : Buffer.from(String(payload), 'utf8');
  const sigBuf = Buffer.from(signatureBase64url, 'base64url');
  if (sigBuf.length !== 64) return false;   // Ed25519 = always 64 bytes
  return crypto.verify(null, buf, publicKey, sigBuf);
}

2. Key Management & URL Fragment Architecture

The decryption key is embedded in the URL fragment (#key=...). Per RFC 3986, the fragment identifier is never sent to the server in HTTP requests. This is the architectural foundation of Vertifile's zero-knowledge guarantee.

2.2 URL Masking in Viewer

When a recipient opens the shared link, the PVF viewer immediately extracts the key from the fragment and then removes it from the address bar using history.replaceState(). A sessionStorage backup allows same-tab refresh without re-sharing the URL.

File: templates/pvf.js — Lines 415-437
var __zkKey = null;
(function() {
  var hash = window.location.hash;
  if (hash && hash.indexOf('#key=') === 0) {
    __zkKey = hash.substring(5);               // Extract base64url key
    // Remove key from address bar and browser history
    if (window.history && window.history.replaceState) {
      window.history.replaceState(null, '',
        window.location.pathname + window.location.search);
    }
    // SessionStorage backup (survives same-tab refresh)
    try {
      sessionStorage.setItem('vf_zk_' + window.location.pathname, __zkKey);
    } catch(e) {}
  } else {
    // Page refresh fallback — recover from sessionStorage
    try {
      __zkKey = sessionStorage.getItem('vf_zk_' + window.location.pathname);
    } catch(e) {}
  }
})();

2.3 Decryption in Public Viewer

The PVF document contains self-contained decryption primitives (no external dependencies). The viewer imports the key, decrypts the blob, verifies the hash, and renders the content.

File: templates/pvf.js — Lines 667-700
// Base64url -> ArrayBuffer
function zkBase64urlToBuffer(b64url) {
  var b64 = b64url.replace(/-/g, '+').replace(/_/g, '/');
  var padded = b64 + '='.repeat((4 - b64.length % 4) % 4);
  var binary = atob(padded);
  var bytes = new Uint8Array(binary.length);
  for (var i = 0; i < binary.length; i++) bytes[i] = binary.charCodeAt(i);
  return bytes.buffer;
}

async function zkImportKey(b64url) {
  var raw = zkBase64urlToBuffer(b64url);
  return crypto.subtle.importKey('raw', raw, { name: 'AES-GCM' }, false, ['decrypt']);
}

async function zkDecrypt(key, ciphertext, iv) {
  return crypto.subtle.decrypt(
    { name: 'AES-GCM', iv: new Uint8Array(iv) }, key, ciphertext);
}

Key Lifecycle: The decryption key exists only (1) in the URL fragment, (2) briefly in JavaScript memory during decryption, (3) in sessionStorage for tab-refresh resilience. After decryption, all copies are zeroed.

3.4 Proof: Server Cannot Access Document Content

The server's involvement in the ZK flow is limited to:

What the server receivesWhat the server does NOT receive
Encrypted blob (AES-256-GCM ciphertext)Plaintext document content
SHA-256 hash of original (hex string)AES-256-GCM decryption key
IV (12-byte initialization vector)URL fragment (#key=...)
Original MIME type, filenameAny data that could reconstruct the key

The pvf_content column in the database stores the complete PVF HTML document, which contains the encrypted base64 blob inside a <script id="encryptedDoc"> tag. Without the AES-256-GCM key (which never touches the server), this blob is cryptographically indistinguishable from random data.

The upload endpoint (routes/user.js:206-318) validates only the encrypted flag, hash format, IV format, and MIME type. It never attempts to read or process the file content.

4. File Handling — PDF Rendering & Pipeline

4.1 Unified PVF Creation Pipeline

All PVF documents — whether created via the API or the dashboard — pass through a single pipeline with 21 ordered steps.

File: services/pvf-pipeline.js — Lines 1-43
1. Validate input (MIME allowlist, non-empty buffer)
2. Hash raw bytes — BLIND processing (never reads content)
3. HMAC sign the hash
4. Generate session token + recipient binding
5. Encode file (text → HTML escape, binary → base64)
6. Fetch branding (org custom logo, colors, waves)
7. Generate shareId BEFORE HTML generation (SHAREID FIX)
8. generatePvfHtml(... shareId) — baked pre-obfuscation
9. DB insert (createDocument)
10. setDocumentUserId + updateUserDocCount
11. incrementDocCount (API key requests)
12. Obfuscate PVF (worker thread)
13. Extract codeIntegrity (SHA-256 of obfuscated JS)
14. Compute chainToken: sha256(hash + sig + orgId + codeIntegrity)
15. Save codeIntegrity + chainedToken to DB
16. Set shareId on documents row
17. Save full obfuscated HTML to pvf_content
18. markDocumentPreviewOnly (free-plan gating)
19. Blockchain registration (fire-and-forget, optional)
20. Audit log entry
21. Return structured result

function appI18nPatch() {
  // 1. Apply static data-i18n attributes
  document.querySelectorAll('[data-i18n]').forEach(function(el) {
    var val = _t(el.getAttribute('data-i18n'));
    if (val && val !== el.getAttribute('data-i18n')) {
      if (/<[a-z][\s\S]*>/i.test(val)) el.innerHTML = val;
      else el.textContent = val;
    }
  });

  // 2. Re-render all dynamic dashboard components
  if (window.Dashboard) {
    if (Dashboard.renderWelcomeDate) Dashboard.renderWelcomeDate();
    if (Dashboard.renderDocRows) Dashboard.renderDocRows();
    if (Dashboard.updatePlanDisplay) Dashboard.updatePlanDisplay();
    if (Dashboard.updateGettingStartedProgress)
      Dashboard.updateGettingStartedProgress();
    if (Dashboard.renderStamps) Dashboard.renderStamps();
  }
}
